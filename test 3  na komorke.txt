<script>

const SHEET_ID =
"1aidR9y-TH5uAU_JFjeCuTDGhnDExpMVefn45AhW0UdM";

const url =
`https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:json`;

let videos = [];
let showingFavs = false;

const container =
document.getElementById("videos");

fetch(url)
.then(res => res.text())
.then(text => {

const json =
JSON.parse(
text.substring(47).slice(0,-2)
);

const rows =
json.table.rows;

videos = rows.map(r => ({

title:
r.c[0]?.v || "",

category:
r.c[1]?.v || "",

parseInt(r.c[4]?.v || 0),

r.c[3]?.v || "0:00",

r.c[7]?.v || "",

r.c[8]?.v || ""

}))

.sort((a,b) => b.order - a.order);

render(videos);

});

function render(list){

container.innerHTML = "";

const favs =
JSON.parse(
localStorage.getItem("favs") || "[]"
);

list.forEach(video => {

const card =
document.createElement("div");

const liked =
favs.includes(video.youtube);

card.className = "card";

card.innerHTML = `

});

}

function toggleFav(link){

let favs =
JSON.parse(
localStorage.getItem("favs") || "[]"
);

if(favs.includes(link)){

favs =
favs.filter(f => f !== link);

}else{

favs.push(link);

}

localStorage.setItem(
"favs",
JSON.stringify(favs)
);

if(showingFavs){

showFavs();

}else{

render(videos);

}

}

function showFavs(){

showingFavs = true;

const favs =
JSON.parse(
localStorage.getItem("favs") || "[]"
);

const filtered =
videos.filter(v =>
favs.includes(v.youtube)
);

render(filtered);

}

function showAll(){

showingFavs = false;

render(videos);

}

function timeToSeconds(t){
if(!t || typeof t !== "string") return 0;
const parts = t.replace(/[^\d:]/g,"").split(":").map(Number);
if(parts.length === 2) return (parts[0]||0)*60+(parts[1]||0);
if(parts.length === 3) return (parts[0]||0)*3600+(parts[1]||0)*60+(parts[2]||0);
return 0;
}

function buildDiscs(selected){
const maxTime = 50*60;
const discs = [];
selected.forEach(v => {
const sT = timeToSeconds(v.duration);
let placed = false;
for(let i = 0; i < discs.length; i++){
const used = discs[i].reduce((acc,x) => acc+timeToSeconds(x.duration),0);
if(used+sT <= maxTime){ discs[i].push(v); placed=true; break; }
}
if(!placed) discs.push([v]);
});
return discs;
}

function getFavMessage(){

const favs =
JSON.parse(
localStorage.getItem("favs") || "[]"
);

const selected =
videos.filter(v =>
favs.includes(v.youtube)
);

const discs = buildDiscs(selected);

let lines = "Szczęść Boże Melodio Wiary!%0ANiech będzie pochwalony Jezus Chrystus.%0A%0AJestem zainteresowany/a płytami z tą muzyką:%0A%0A";

discs.forEach((disc, idx) => {
lines += "PŁYTA NUMER " + (idx+1) + ":%0A";
disc.forEach(v => {
lines += "• " + v.order + " - " + v.title + "%0A";
});
lines += "%0A";
});

return lines;

}
